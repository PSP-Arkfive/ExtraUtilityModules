target_sources(kbooti_update PRIVATE
   main.c
)

target_link_libraries(kbooti_update PRIVATE
   -nostartfiles
   -nostdlib
   pspsystemctrl_kernel
   pspsdk
   pspmodinfo
   pspkernel
)
